This guide will get you up and running with SlateDB quickly. We'll create a simple key-value store using an in-memory object store for simplicity.

## Choose Your Language

<Tabs>
  <Tab title="Rust">
    <Steps>
      <Step title="Add Dependencies">
        Add SlateDB to your `Cargo.toml`:

        ```toml Cargo.toml
        [dependencies]
        slatedb = "*"
        tokio = { version = "1", features = ["full"] }
        ```
      </Step>

      <Step title="Write Your First Program">
        Create a new file `main.rs` with the following code:

        ```rust main.rs
        use slatedb::{Db, Error};
        use slatedb::object_store::memory::InMemory;
        use std::sync::Arc;

        #[tokio::main]
        async fn main() -> Result<(), Error> {
            // Create an in-memory object store
            let object_store = Arc::new(InMemory::new());
            
            // Open the database
            let db = Db::open("/tmp/my_slatedb", object_store).await?;

            // Put a key-value pair
            db.put(b"hello", b"world").await?;

            // Get the value back
            let value = db.get(b"hello").await?;
            println!("Value: {:?}", value); // Some(b"world")

            // Delete the key
            db.delete(b"hello").await?;
            assert!(db.get(b"hello").await?.is_none());

            // Close the database
            db.close().await?;

            Ok(())
        }
        ```
      </Step>

      <Step title="Run It">
        ```bash
        cargo run
        ```
      </Step>
    </Steps>
  </Tab>

## Working with Range Scans

One of SlateDB's powerful features is efficient range scanning. Here's how to iterate over keys:

<CodeGroup>
```rust Rust
use slatedb::{Db, Error};
use slatedb::object_store::memory::InMemory;
use std::sync::Arc;

#[tokio::main]
async fn main() -> Result<(), Error> {
    let object_store = Arc::new(InMemory::new());
    let db = Db::open("/tmp/my_slatedb", object_store).await?;

    // Insert multiple keys
    db.put(b"user:1", b"Alice").await?;
    db.put(b"user:2", b"Bob").await?;
    db.put(b"user:3", b"Charlie").await?;
    db.put(b"product:1", b"Widget").await?;

    // Scan all keys with "user:" prefix
    let mut iter = db.scan(b"user:"..).await?;
    while let Ok(Some(kv)) = iter.next().await {
        println!("{:?} = {:?}", kv.key, kv.value);
    }

    // Scan bounded range
    let mut iter = db.scan(b"user:1"..=b"user:2").await?;
    while let Ok(Some(kv)) = iter.next().await {
        println!("{:?} = {:?}", kv.key, kv.value);
    }

    // Seek to specific key in iterator
    let mut iter = db.scan::<Vec<u8>, _>(..).await?;
    iter.seek(b"user:2").await?;
    if let Ok(Some(kv)) = iter.next().await {
        println!("Found: {:?} = {:?}", kv.key, kv.value);
    }

    db.close().await?;
    Ok(())
}
```

```go Go
package main

import (
    "fmt"
    "io"
    "slatedb.io/slatedb-go"
)

func main() {
    db, _ := slatedb.Open(
        "/tmp/my_slatedb",
        slatedb.WithUrl[slatedb.DbConfig]("memory:///"),
    )
    defer db.Close()

    // Insert multiple keys
    db.Put([]byte("user:1"), []byte("Alice"))
    db.Put([]byte("user:2"), []byte("Bob"))
    db.Put([]byte("user:3"), []byte("Charlie"))

    // Scan with prefix
    iter, _ := db.ScanPrefix([]byte("user:"))
    defer iter.Close()

    for {
        kv, err := iter.Next()
        if err == io.EOF {
            break
        }
        if err != nil {
            panic(err)
        }
        fmt.Printf("%s = %s\n", kv.Key, kv.Value)
    }
}
```

```java Java
import io.slatedb.*;
import java.nio.charset.StandardCharsets;

public class ScanExample {
    public static void main(String[] args) throws Exception {
        SlateDb.initLogging(SlateDbConfig.LogLevel.INFO);

        try (SlateDb db = SlateDb.open("/tmp/my_slatedb", "memory:///", null)) {
            // Insert multiple keys
            db.put("user:1".getBytes(), "Alice".getBytes());
            db.put("user:2".getBytes(), "Bob".getBytes());
            db.put("user:3".getBytes(), "Charlie".getBytes());

            // Scan by prefix
            try (SlateDbScanIterator iter = db.scanPrefix("user:".getBytes())) {
                SlateDbKeyValue kv;
                while ((kv = iter.next()) != null) {
                    System.out.println(
                        new String(kv.key(), StandardCharsets.UTF_8) + " = " +
                        new String(kv.value(), StandardCharsets.UTF_8)
                    );
                }
            }
        }
    }
}
```

## Using Real Object Storage

The in-memory object store is great for testing, but for production you'll want to use real object storage. Here's an example with Amazon S3:

<CodeGroup>
```rust Rust
use slatedb::{Db, Error};
use object_store::aws::{AmazonS3Builder, S3ConditionalPut};
use std::sync::Arc;

#[tokio::main]
async fn main() -> Result<(), Error> {
    // Configure S3 object store
    let object_store = Arc::new(
        AmazonS3Builder::new()
            .with_bucket_name("my-slatedb-bucket")
            .with_region("us-east-1")
            .with_conditional_put(S3ConditionalPut::ETagMatch)
            .build()?
    );

    // Open database with S3 backend
    let db = Db::open("/tmp/slatedb_cache", object_store).await?;

    // Use the database as before
    db.put(b"key", b"value").await?;
    let value = db.get(b"key").await?;
    
    db.close().await?;
    Ok(())
}
```

```python Python
from slatedb import SlateDB

# Using S3 with environment variables (AWS credentials)
db = SlateDB(
    "/tmp/slatedb_cache",
    url="s3://my-slatedb-bucket/"
)

# Use the database as before

value = db.get(b"key")

db.close()
```

package main

import "slatedb.io/slatedb-go"
